36    pub fn build_game_context_prompt(game_state: &GameState) -> String {
37        format!(
38            "GAME CONTEXT:\n\
39            Player Name: {}\n\
40            Current Location: {}\n\
41            Inventory: {:?}\n\
42            Visited Locations: {:?}\n\n\
43            STORY SUMMARY:\n\
44            {}\n\n\
45            LAST AI RESPONSE:\n\
46            {}\n\n\
47            CURRENT PLAYER ACTION:\n\
48            {}\n\n\
49            Please continue the story based on the player's action and respond with the JSON format specified in the system prompt.",
50            game_state.name,
51            game_state.current_location,
52            game_state.inventory,
53            game_state.visited_locations,
54            game_state.summary,
55            game_state.last_response.as_ref().unwrap_or(&"[Game Start]".to_string()),

71    pub fn get_theme_selection_prompt() -> String {
72        let title = "Select the theme of your adventure:".cyan().bold();
73        let options = vec![
74            "1. Fantasy - Dragons, magic, and enchanted realms".green(),
75            "2. Dark/Horror - Cosmic horror and dark mysteries".red(),
76            "3. Sci-Fi - Space, technology, and aliens".blue(),
77            "4. Historical - Swords, duels, and intrigue".yellow(),
78            "5. Surprise me - Let the AI choose for you".magenta(),
79            "6. You choose - Custom theme".cyan()
80        ];
81
82        let mut prompt = format!("{}\n", title);
83        for option in options {
84            prompt.push_str(&format!("{}\n", option));
85        }
86        prompt.push_str("Please enter the number of your choice: ");
87        prompt
88    }
